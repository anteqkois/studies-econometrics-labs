Zadanie 3. Metoda Hellwiga

3.1

Na podstawie metody Hellwiga doboru zmiennych do modelu znale optymalny podzbir zmiennych objaniajcych. W tym celu naley rozway wszystkie moliwe kombinacje potencjalnych zmiennych objaniajcych (mileage, tax, engineSize) i wyznaczy dla nich indywidualne, a nastpnie integralne wskaniki pojemnoci informacyjnej

import itertools
import numpy as np

# Dane (macierz korelacji obliczona wczeniej w 2.1)
correlation_matrix = pd.DataFrame({
    "price": [1.000000, -0.535357, 0.356157, 0.591262],
    "mileage": [-0.535357, 1.000000, -0.166547, 0.070710],
    "tax": [0.356157, -0.166547, 1.000000, 0.393075],
    "engineSize": [0.591262, 0.070710, 0.393075, 1.000000]
}, index=["price", "mileage", "tax", "engineSize"])

# Zmienna zalena i zbir zmiennych objaniajcych
y = "price"
X = ["mileage", "tax", "engineSize"]

# Wszystkie moliwe kombinacje zmiennych objaniajcych
subsets = []
for r in range(1, len(X) + 1):
    subsets.extend(itertools.combinations(X, r))

hellwig_indices = {}

for subset in subsets:
    subset = list(subset)
    
    # Wektor korelacji zmiennych objaniajcych z price
    r_yx = correlation_matrix.loc[y, subset].values
    
    # Macierz korelacji midzy zmiennymi objaniajcymi
    R_xx = correlation_matrix.loc[subset, subset].values
    
    # Wskanik Hellwiga H
    try:
        R_xx_inv = np.linalg.inv(R_xx)  # Odwracamy macierz korelacji zmiennych objaniajcych
        h = np.sum((r_yx @ R_xx_inv) * r_yx) / len(subset)  # Obliczamy wskanik Hellwiga
        hellwig_indices[tuple(subset)] = h
    except np.linalg.LinAlgError:
        pass  # Jeli macierz nie jest odwracalna, pomijamy ten przypadek

# Znalezienie najlepszego podzbioru zmiennych
best_subset = max(hellwig_indices, key=hellwig_indices.get)
best_hellwig_index = hellwig_indices[best_subset]

# Wyniki
print("Najlepszy podzbir zmiennych objaniajcych:", best_subset)
print("Najwyszy wskanik Hellwiga:", best_hellwig_index)
